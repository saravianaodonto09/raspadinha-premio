<script>

const canvas = document.getElementById("scratch");
const ctx = canvas.getContext("2d");

let revelado = false;
let desenhando = false;

function desenharCamada(){

    const rect = canvas.getBoundingClientRect();

    canvas.width = rect.width;
    canvas.height = rect.height;

    ctx.globalCompositeOperation = "source-over";

    const grad = ctx.createLinearGradient(
        0,
        0,
        canvas.width,
        canvas.height
    );

    grad.addColorStop(0,"#d9d9d9");
    grad.addColorStop(.5,"#bdbdbd");
    grad.addColorStop(1,"#9f9f9f");

    ctx.fillStyle = grad;

    ctx.fillRect(
        0,
        0,
        canvas.width,
        canvas.height
    );

    for(let i=0;i<250;i++){

        ctx.fillStyle =
        `rgba(255,255,255,${
            Math.random()*0.25
        })`;

        ctx.fillRect(
            Math.random()*canvas.width,
            Math.random()*canvas.height,
            2,
            2
        );
    }

    ctx.fillStyle="#666";

    ctx.textAlign="center";

    ctx.font=`bold ${
        canvas.width*0.08
    }px Arial`;

    ctx.fillText(
        "RASPE AQUI",
        canvas.width/2,
        canvas.height/2
    );

    ctx.globalCompositeOperation =
    "destination-out";
}

desenharCamada();

window.addEventListener(
    "resize",
    desenharCamada
);

function raspar(x,y){

    ctx.beginPath();

    ctx.arc(
        x,
        y,
        25,
        0,
        Math.PI*2
    );

    ctx.fill();

    verificar();
}

function verificar(){

    if(revelado) return;

    const pixels =
    ctx.getImageData(
        0,
        0,
        canvas.width,
        canvas.height
    ).data;

    let transparentes = 0;

    for(
        let i=3;
        i<pixels.length;
        i+=4
    ){

        if(
            pixels[i] === 0
        ){
            transparentes++;
        }
    }

    const percentual =
    transparentes /
    (pixels.length/4);

    if(percentual > 0.45){

        revelado = true;

        confetti({

            particleCount:250,

            spread:120,

            origin:{
                y:.6
            }

        });

        if(
            navigator.vibrate
        ){

            navigator.vibrate([
                200,
                100,
                200
            ]);

        }

        const audio =
        document.getElementById(
            "vitoria"
        );

        audio.play().catch(()=>{});

        canvas.style.transition =
        ".8s";

        canvas.style.opacity = "0";

        canvas.style.pointerEvents =
        "none";
    }
}

function posicao(e){

    const rect =
    canvas.getBoundingClientRect();

    if(e.touches){

        return{

            x:
            e.touches[0].clientX
            - rect.left,

            y:
            e.touches[0].clientY
            - rect.top

        };
    }

    return{

        x:
        e.clientX
        - rect.left,

        y:
        e.clientY
        - rect.top

    };
}

canvas.addEventListener(
    "mousedown",
    ()=>{
        desenhando=true;
    }
);

canvas.addEventListener(
    "mouseup",
    ()=>{
        desenhando=false;
    }
);

canvas.addEventListener(
    "mousemove",
    (e)=>{

        if(!desenhando) return;

        const p =
        posicao(e);

        raspar(
            p.x,
            p.y
        );
    }
);

canvas.addEventListener(
    "touchstart",
    ()=>{
        desenhando=true;
    }
);

canvas.addEventListener(
    "touchend",
    ()=>{
        desenhando=false;
    }
);

canvas.addEventListener(
    "touchmove",
    (e)=>{

        e.preventDefault();

        if(
            !desenhando
        ) return;

        const p =
        posicao(e);

        raspar(
            p.x,
            p.y
        );

    },
    {
        passive:false
    }
);
